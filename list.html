// List operations
#include <stdio.h>
#include <conio.h>
#define max 20
int top = -1;
int a[10];
void push();
void pop();
void display();
void palindrome();

void main()
{
int ch;
while (1)
{
printf("\n1.Push\n2.Pop\n3.Display\n4.Palindrome\nEnter number: ");
scanf("%d", &ch);
switch (ch)
{
case 1:
{
push();
break;
}
case 2:
{
pop();
break;
}
case 3:
{
display();
break;
}
case 4:
{
palindrome();
break;
}
default:
{
exit(0);
}
}
}
}
void push()
{
int item;
if (top >= max)
{
printf("\nStack Overflow");
}
else
{
top += 1;
printf("Enter element: ");
scanf("%d", &item);
a[top] = item;
}
}
void pop()
{
int item;
if (top < 0)
{
printf("Stack Underflow");
}
else
{
item = a[top];
top -= 1;
printf("\nValue deleted is %d", item);
}
}
void display()
{
int i;
if (top < 0)
{
printf("Stack Underflow");
}
else
{
for (i = 0; i <= top; i++)
{
printf("\n%d", a[i]);
}
}
}
void palindrome()
{
int i, temp;
temp = top;
if (top != -1)
{
for (i = 0; i <= top / 2; i++)
{
if (a[i] != a[temp--])
{
printf("The stack is not palindrome");
return;
}
printf("the stack is palindrome");
}
return;
}
}
