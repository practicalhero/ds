#include <stdio.h>
#include <string.h>
char infix[50], suffix[50], stack[25];
int top = -1;
void convert_fun();
int precedence(char);
void push(char);
char pop();
int main()
{
printf("\n\tenter infix exp");
scanf("%s", infix);
convert_fun();
printf("the suffix exp is:%s", suffix);
}
void convert_fun()
{
int i = 0, j = 0;
char c, temp;
push('#');
while (i < strlen(infix))
{
c = infix[i++];
switch (c)
{
case '(':
push(c);
break;
case ')':
{
temp = pop();
while (temp != 'c')
{
suffix[j++] = temp;
temp = pop();
}
break;
}

case '+':
case '-':
case '*':
case '/':
case '^':
while (precedence(stack[top]) >= precedence(c))
{
temp = pop();
suffix[j++] = temp;
}
push(c);
break;

default:
{
suffix[j++] = c;
break;
}
}
}
while (top > 0)
{
temp = pop();
suffix[j++] = temp;
}
}
void push(char c)
{
stack[++top] = c;
}

char pop()
{
return (stack[top--]);
}

int precedence(char c)
{
switch (c)
{
case '^':
return (3);
case '/':
return (2);
case '+':
return (1);
case '-':
return (1);
case '(':
return (0);
case ')':
return (0);
case '#':
return (-1);
}
return (0);
}
