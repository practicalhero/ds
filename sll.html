void create(int us, int se, int ph, char na[], char br[])
{ struct node *q, *tmp;

tmp->usn = us;
tmp->sem = se;
tmp->phno = ph;
strcpy(tmp->name, na);
strcpy(tmp->branch, br);
tmp->link = NULL;
if (start == NULL)
{
start = tmp;
}
else
{
q = start;
while (q->link != NULL)
{
q = q->link;
}
q->link = tmp;
}
}
